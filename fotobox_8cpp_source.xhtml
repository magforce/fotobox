143 void FotoBox::keyPressEvent(QKeyEvent *event)
144 {
145  //prevent triggering method too often
146  if (!event->isAutoRepeat()) {
147  //start FotoBox
148  if (std::any_of(m_triggerKey.cbegin(), m_triggerKey.cend(), [&](const Qt::Key i_key) { return event->key() == i_key; })) {
149  Q_EMIT start();
150  return;
151  }
152 
153  //preferences dialog
154  if (std::any_of(m_preferenceKey.cbegin(), m_preferenceKey.cend(), [&](const Qt::Key i_key) { return event->key() == i_key; })) {
155  preferenceDialog();
156  return;
157  }
158 
159  //quit application
160  if (std::any_of(m_quitKey.cbegin(), m_quitKey.cend(), [&](const Qt::Key i_key) { return event->key() == i_key; })) {
161  if (event->modifiers() == Qt::ShiftModifier) {
162  QCoreApplication::quit();
163  }
164  m_ui->statusBar->showMessage(tr("To quit the application, please hold down the Shift key while press Escape key."), STATUSBAR_MSG_TIMEOUT);
165  return;
166  }
167  }
168 }

230 void FotoBox::loadPhoto(const QString &i_filePath)
231 {
232  //try to load the photo shot by camera
233  if (m_photo.load(i_filePath)) {
234  //resize picture to label size
235  QSize size(m_ui->lblPhoto->width(), m_ui->lblPhoto->height());
236 
237  //show photo in grayscale (monochrome photography)
238  if (PreferenceProvider::instance().grayscale()) {
239  const auto grey = m_photo.toImage().convertToFormat(QImage::Format_Grayscale8);
240  m_photo.convertFromImage(grey);
241  }
242 
243  //scale photo and show it immediately
244  m_photo = m_photo.scaled(size, Qt::KeepAspectRatio, Qt::SmoothTransformation);
245  m_ui->lblPhoto->setPixmap(m_photo);
246  m_ui->lblPhoto->repaint();
247  } else {
248  m_ui->statusBar->showMessage(tr("Couldn't load the photo."), STATUSBAR_MSG_TIMEOUT);
249  }
